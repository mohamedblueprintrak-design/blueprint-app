                    "message": prompt or "",
                    "project_id": project_id,
                    "history": json.dumps(st.session_state.msgs)
                }
                try:
                    r = requests.post(f"{BACKEND}/process", data=data_payload, files=files, timeout=60)
                    
                    if r.ok:
                        res = r.json()
                        ans_text = ""
                        
                        # عرض نتائج المساعد
                        results = res.get("results", {})
                        for k, v in results.items():
                            if k == "image_data":

                                )

                            elif "تحذير" in k or "warning" in k.lower():

                            else:
                                st.markdown(f"**{k}**")
                                st.markdown(v)

                                ans_text += v + "\n"
                        
                        # حفظ رد المساعد

                            st.session_state.msgs.append({"role": "assistant", "content": ans_text})

                        st.error(_("❌ فشل في الاتصال بالخادم", "❌ Failed to connect to server"))

                    st.error(f"⚠️ {_('خطأ', 'Error')}: {str(e)}")

@app.post("/process")
async def process_request(
    message: str = Form(None), 
    project_id: int = Form(...), 
    file: UploadFile = File(None), 
    history: str = Form("[]")
):
    # 1. قراءة الملف
    file_bytes = await file.read() if file else None
    file_type = file.content_type if file else None
    
    db = SessionLocal()
    file_record = None
    
    # 2. حفظ الملف (لو موجود) في قاعدة البيانات والديسك
    try:
        if file_bytes and file:
            # حفظ على الديسك
            file_path = os.path.join(UPLOAD_DIR, f"{project_id}_{file.filename}")
            with open(file_path, "wb") as f:
                f.write(file_bytes)
            
            # حفظ في الـ DB
            file_record = UploadedFile(
                project_id=project_id, 
                filename=file.filename, 
                file_type=file_type, 
                file_path=file_path,
                analyzed=True
            )
            db.add(file_record)
            db.commit()
            db.refresh(file_record)
            logger.info(f"File saved: {file.filename}")
    except Exception as e:
        logger.error(f"File saving error: {e}")
    finally:
        db.close()  # نغلق الجلسة بعد حفظ الملف

    # 3. تشغيل العقل المدبر (Orchestrator) - Async
    result = await route_request(message, file_bytes, file_type, project_id, json.loads(history))
    
    # 4. حفظ نتائج الـ Orchestrator في قاعدة البيانات (نحتاج جلسة جديدة)
    db = SessionLocal()
    try:
        domain = result.get("domain")
        results = result.get("results", {})
        
        # حفظ الحصر (BOQ)
        if domain == "boq" and "boq_item" in results:
            item = results["boq_item"]
            new_item = BOQItem(
                project_id=project_id,
                description=item.get("description"),
                unit=item.get("unit"),
                quantity=item.get("quantity"),
                total_price=item.get("total_price", 0)
            )
            db.add(new_item)
        
        # حفظ التحليلات (Design)
        elif domain == "design":
            new_analysis = Analysis(
                project_id=project_id,
                task_type=result.get("task"),
                input_text=message,
                result_json=result,
                safety_status="Unsafe" if "⚠️ تحذير" in str(results) else "Safe"
            )
            db.add(new_analysis)
        
        # حفظ العيوب (Site Defects)
        elif domain == "site":
            # حفظ تقرير الموقع
            if "checklist_data" in results:
                report = SiteReport(
                    project_id=project_id,
                    summary=message,
                    issues=json.dumps(results["checklist_data"])
                )
                db.add(report)
            
            # حفظ العيوب المكتشفة بالـ Vision
            if "⚠️ العيوب" in results:
                new_defect = Defect(
                    project_id=project_id,
                    description=results["⚠️ العيوب"],
                    severity="Medium", # يمكن تحديدها بـ AI لاحقاً
                    status="Open",
                    image_id=file_record.id if file_record else None
                )
                db.add(new_defect)
        
        # حفظ ردود الدردشة العادية في timeline (اختياري)
        elif domain == "general" and message:
            # يمكن حفظها كتحليل عام
            new_analysis = Analysis(
                project_id=project_id,
                task_type="chat",
                input_text=message,
                result_json=result,
                safety_status="Safe"
            )
            db.add(new_analysis)
                
        db.commit()
        logger.info(f"Data saved for Project {project_id} - Domain: {domain}")
        
    except Exception as e:
        logger.error(f"Database Save Error: {e}")
        db.rollback()
    finally:
        db.close()
        
    return result

def structural_beam_analysis(length, load):
    try:
        L = length; w = load * 10; M = (w * L**2) / 8
        scenarios = [
            _calc_beam(L, M, 250, int((L/12.0)*1000), "💰 Economic"),
            _calc_beam(L, M, 250, int((L/18.0)*1000), "🪶 Lightweight"),
            _calc_beam(L, M, 300, int((L/10.0)*1000), "🛡️ Safe")
        ]
        return {"success": True, "scenarios": scenarios}
    except Exception as e: return {"success": False, "error": str(e)}

def _calc_beam(L, M, w, d, label):
    d_eff = d - 40
    if d_eff <= 0: d_eff = 100
    As_req = (M * 1e6) / (0.87 * 360 * d_eff)
    bars = math.ceil(As_req / 201)
    
    vol = (w/1000) * (d/1000) * L
    steel_w = (201 * bars * L * 7.85) / 1000000
    cost = (vol * UNIT_PRICES['concrete_m3']) + (steel_w * UNIT_PRICES['steel_ton'])
    
    return {
        "label": label, "width_mm": w, "depth_mm": d,
        "steel_bars": f"{bars} T16", "cost_egp": round(cost),
        "raw_values": {"type": "beam", "depth_mm": d, "length_m": L, "steel_ratio_percent": round((As_req/(w*d_eff))*100, 2)}
    }

def slab_analysis(length_short, length_long, thickness=0):
    try:
        ratio = length_long / length_short
        type_str = "Two-Way" if ratio < 2 else "One-Way"
        t = thickness if thickness > 0 else (length_short * 1000 / 30)
        
        # حساب تقريبي للحديد
        load = 15.0 # kN/m2 approx
        M = (load * length_short**2) / 8 # Simplified
        As_req = (M * 1e6) / (0.87 * 360 * (t-20))
        
        return {
            "success": True,
            "results": {
                "explanation": f"🔲 **بلاطة {type_str}**: السمك المطلوب {t:.0f} مم، الحديد التقريبي {As_req/201:.2f} قضيب/متر",
                "raw_values": {"type": "slab", "thickness_mm": t}
            }
        }
    except Exception as e: return {"success": False, "error": str(e)}

def column_analysis(axial_load, height):
    try:
        # تصميم عمود بسيط
        Pu = axial_load * 1.5 * 10 # Factored Load
        Ag = Pu / (0.4 * 30) # Approx Area
        side = math.sqrt(Ag) * 1000
        
        return {
            "success": True,
            "results": {
                "explanation": f"🏗️ **عمود**: المقاس المقترح {side:.0f}x{side:.0f} مم للحمل {axial_load} ك.ن",
                "raw_values": {"type": "column"}
            }
        }
    except Exception as e: return {"success": False, "error": str(e)}

def foundation_analysis(load, soil_capacity):
    try:
        # تصميم قاعدة منفصلة
        area = (load * 1.5) / (soil_capacity * 10) # m2
        side = math.sqrt(area)
        thickness = side * 0.5 # Approx thickness
        
        return {
            "success": True,
            "results": {
                "explanation": f"🧱 **قاعدة**: الأبعاد {side:.2f}x{side:.2f} متر، سمك {thickness*100:.0f} سم",
                "raw_values": {"type": "foundation"}
            }
        }
    except Exception as e: return {"success": False, "error": str(e)}

def retaining_wall_analysis(height, soil_load):
    """
    تصميم مبسط لجدار استنادي
    height: ارتفاع الجدار بالمتر
    soil_load: حمل التربة (kN/m2) تقديري
    """
    try:
        # سمك القاعدة غالباً = 0.1 * الارتفاع + 0.3
        base_thickness = 0.1 * height + 0.3
        # عرض القاعدة = 0.5 * الارتفاع
        base_width = 0.5 * height
        # تسليح تقريبي
        steel_main = f"16mm كل 200 مم"
        return {
            "success": True,
            "results": {
                "explanation": f"🧱 **جدار استنادي**: ارتفاع {height} م، عرض القاعدة {base_width:.2f} م، سمك القاعدة {base_thickness:.2f} م، تسليح رئيسي {steel_main}",
                "raw_values": {"type": "retaining_wall", "height": height, "base_width": base_width, "base_thickness": base_thickness}
            }
        }
    except Exception as e:
        return {"success": False, "error": str(e)}

# ---------- إضافة جديدة: تصميم سلم ----------
def stair_analysis(floor_height, riser=0.17, tread=0.30):
    """
    تصميم سلم خرساني
    floor_height: ارتفاع الدور (متر)
    riser: ارتفاع القائمة (متر)
    tread: عرض النائمة (متر)
    """
    try:
        number_of_risers = floor_height / riser
        number_of_treads = number_of_risers - 1
        flight_length = number_of_treads * tread
        # سمك السلم تقديري
        thickness = flight_length * 0.03  # تقريبي
        return {
            "success": True,
            "results": {
                "explanation": f"🪜 **سلم**: عدد القوائم {number_of_risers:.0f}، طول الباي {flight_length:.2f} م، السمك التقريبي {thickness*100:.0f} مم",
                "raw_values": {"type": "stair", "risers": number_of_risers, "flight_length": flight_length, "thickness_mm": thickness*1000}
            }
        }
    except Exception as e:
        return {"success": False, "error": str(e)}

class LLMProvider:
    def __init__(self):
        self.models = {
            "gpt-4o": {"type": "openai", "model": "gpt-4o", "client": OPENAI_CLIENT},
            "gpt-3.5": {"type": "openai", "model": "gpt-3.5-turbo", "client": OPENAI_CLIENT},
            "deepseek": {"type": "openai_compat", "base_url": "https://api.deepseek.com", "model": "deepseek-chat"},
            "grok": {"type": "openai_compat", "base_url": "https://api.x.ai/v1", "model": "grok-beta"},
            "gemini": {"type": "gemini", "model": "gemini-1.5-flash"},
            "zai": {"type": "openai_compat", "base_url": "https://api.z.ai/v1", "model": "zai-chat"}
        }

    async def _get_client(self, model_key):
        config = self.models.get(model_key)
        if not config: return None, None

        if config["type"] in ["openai", "openai_compat"]:
            if config.get("client"): 
                return config["client"], config["model"]
            else:
                # محاولة إنشاء Client للمفاتيح الخارجية
                key = os.getenv(f"{model_key.upper()}_API_KEY")
                if key:
                    try:
                        client = AsyncOpenAI(api_key=key, base_url=config.get("base_url"))
                        return client, config["model"]
                    except: pass
        return None, None

    async def generate_text(self, prompt, model_preference=["gemini", "gpt-4o", "deepseek", "grok"]):
        """يختار أفضل موديل متاح للرد النصي"""
        
        for model_name in model_preference:
            try:
                if model_name == "gemini":
                    key = os.getenv("GEMINI_API_KEY")
                    if key:
                        model = genai.GenerativeModel('gemini-1.5-flash')
                        # تشغيل المكالمة المتزامنة في خيط منفصل (مهم جداً)
                        res = await asyncio.to_thread(model.generate_content, prompt)
                        return res.text
                
                else:
                    # DeepSeek, Grok, OpenAI, Z.ai
                    client, model_id = await self._get_client(model_name)
                    if client:
                        res = await client.chat.completions.create(
                            model=model_id,
                            messages=[{"role": "user", "content": prompt}]
                        )
                        return res.choices[0].message.content
            except Exception as e:
                logger.warning(f"Model {model_name} failed: {e}")
                continue
        
        return "⚠️ لم أتمكن من الاتصال بأي نموذج ذكي. تأكد من مفاتيح API."

                    key = os.getenv("GEMINI_API_KEY")
                    if key:
                        model = genai.GenerativeModel('gemini-1.5-flash')
                        res = await asyncio.to_thread(
                            model.generate_content,
                            [prompt, {"mime_type": "image/jpeg", "data": base64.b64decode(image_base64)}]
                        )
                        return res.text
                
                elif model_name == "gpt-4o":
                    client, model_id = await self._get_client("gpt-4o")
                    if client:
                        res = await client.chat.completions.create(
                            model="gpt-4o",
                            messages=[{

                                "content": [
                                    {"type": "text", "text": prompt},
                                    {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{image_base64}"}}
                                ]
                            }]
                        )
                        return res.choices[0].message.content
            except Exception as e:
                logger.warning(f"Vision Model {model_name} failed: {e}")
                continue
        
        return "⚠️ لم أتمكن من تحليل الصورة."

def check_beam_safety(results: dict):
    warnings = []
    suggestions = []
    is_safe = True

    steel_ratio = results.get("steel_ratio_percent", 0)
    depth = results.get("depth_mm", 0)
    length = results.get("length_m", 0)

    if steel_ratio > 2.5:
        warnings.append(f"⚠️ تحذير: نسبة التسليح عالية جداً ({steel_ratio}%)، قد تحدث تشققات.")
        suggestions.append("💰 اقتراح: زيادة عمق الكمرة لتقليل نسبة الحديد.")
        is_safe = False
    elif steel_ratio < 0.2:
        warnings.append("⚠️ تحذير: نسبة التسليح منخفضة جداً.")
        suggestions.append("💡 اقتراح: مراجعة الحسابات أو إضافة حديد إضافي.")

    # فحص عمق الكمرة
    if length > 0 and depth > 0:
        depth_span_ratio = depth / (length * 1000)
        if depth_span_ratio < 1/15:
            warnings.append("⚠️ الكمرة نحيفة جداً (نسبة العمق/الطول صغيرة).")
            suggestions.append("📏 اقتراح: زيادة العمق لتحسين المقاومة.")
            is_safe = False

    return {"is_safe": is_safe, "warnings": warnings, "suggestions": suggestions}

        c.drawString(x, y, text)

def generate_project_report(project_id):
    db = SessionLocal()
    try:
        project = db.query(Project).filter(Project.id == project_id).first()
        if not project: return None

        analyses = db.query(Analysis).filter(Analysis.project_id == project_id).all()
        boqs = db.query(BOQItem).filter(BOQItem.project_id == project_id).all()
        defects = db.query(Defect).filter(Defect.project_id == project_id).all()

        filename = f"report_{project_id}.pdf"
        c = canvas.Canvas(filename, pagesize=A4)
        w, h = A4

        # 1. العنوان
        c.setFont(FONT_NAME, 24)
        draw_arabic_text(c, f"تقرير مشروع: {project.name}", 50, h - 50, 24)
        
        c.setFont(FONT_NAME, 12)
        draw_arabic_text(c, f"الموقع: {project.location}", 50, h - 80, 12)
        
        # خط فاصل
        c.line(50, h - 90, w - 50, h - 90)

        y = h - 120

        # 2. قسم التحليلات
        draw_arabic_text(c, "التحليلات الهندسية", 50, y, 16)
        y -= 30

        for ana in analyses:
            if y < 100:
                c.showPage()
                y = h - 50

            # اسم المهمة
            draw_arabic_text(c, f"- مهمة: {ana.task_type}", 50, y, 12)
            y -= 20
            
            # محتوى النتيجة (نص مختصر)
            res_data = ana.result_json.get("results", {}) if ana.result_json else {}
            text_content = res_data.get("📐 النتائج", "") or res_data.get("📄 تحليل الملف", "")
            
            if text_content:
                # نقطع النص عشان ميطولش
                short_text = str(text_content).replace("**", "")[:100] 
                draw_arabic_text(c, short_text, 70, y, 10)
                y -= 20
            
            # صورة الرسم (لو موجودة)
            img_data = res_data.get("image_data")
            if img_data:
                try:
                    img_bytes = base64.b64decode(img_data)
                    img = ImageReader(io.BytesIO(img_bytes))
                    c.drawImage(img, 50, y - 120, width=150, height=100, preserveAspectRatio=True)
                    y -= 140
                except: pass
            
            y -= 20

        # 3. قسم العيوب
        if y < 150:
            c.showPage()
            y = h - 50
        y -= 20
        draw_arabic_text(c, "العيوب المسجلة", 50, y, 16)
        y -= 30
        
        for defect in defects:
            if y < 100:
                c.showPage()
                y = h - 50
            line = f"- {defect.description} | {defect.severity} | {defect.status}"
            draw_arabic_text(c, line, 50, y, 10)
            y -= 20
            
            # إذا كان للعيب صورة، نحاول إدراجها (اختياري)
            if defect.image_id:
                img_file = db.query(UploadedFile).filter(UploadedFile.id == defect.image_id).first()
                if img_file and os.path.exists(img_file.file_path):
                    try:
                        c.drawImage(img_file.file_path, 50, y-80, width=100, height=80, preserveAspectRatio=True)
                        y -= 100
                    except: pass
        y -= 20

        # 4. قسم الكميات
        if y < 150:
            c.showPage()
            y = h - 50

        y -= 20
        draw_arabic_text(c, "جدول الكميات (BOQ)", 50, y, 16)
        y -= 30
        
        total_cost = 0
        for item in boqs:
            price = item.total_price or 0
            total_cost += price
            line = f"{item.description} | {item.quantity} {item.unit} | {price} جنيه"
            draw_arabic_text(c, line, 50, y, 10)
            y -= 20
            
        # الإجمالي
        y -= 10
        c.setFont(FONT_NAME, 14)
        draw_arabic_text(c, f"الإجمالي التقديري: {total_cost} جنيه", 50, y, 14)

        c.save()
        
        with open(filename, "rb") as f:
            pdf_bytes = f.read()
        os.remove(filename)
        return pdf_bytes

    except Exception as e:
        logger.error(f"PDF Gen Error: {e}")
        return None
    finally:
        db.close()

لاحظ أن بعض الحزم قد تحتاج إلى تثبيت إضافي حسب نظامك (مثل arabic-reshaper, bidi). يمكنك استخدام pip install arabic-reshaper python-bidi.

الآن يمكنك نسخ كل ملف على حدة ولصقه في الملف المقابل. إذا واجهتك أي مشكلة، فأنا هنا للمساعدة.

جميل ثبت اخر ملفات  عندك تبقي هيا الاس الي هنطور علية بعد كدا
نحن الآن بعد أن قدمنا جميع الملفات المحدثة، يمكننا اعتبار هذه النسخة هي "الإصدار الأساسي" الذي سنبني عليه التطويرات المستقبلية. لقد أضفنا العديد من الميزات الجديدة دون المساس بالكود الأصلي، وحافظنا على هيكلية المشروع. من الآن فصاعداً، أي تطوير إضافي سيتم إضافته على هذه القاعدة.

إذا كنت تريد توثيق هذه النسخة، يمكنك إنشاء مجلد جديد وحفظ جميع الملفات فيه. أو يمكنك وضع العلامة (tag) في نظام التحكم بالنسخ (Git) إذا كنت تستخدمه.

الآن، أنا جاهز لأي طلبات تطوير إضافية تريدها. فقط أخبرني بالميزة الجديدة التي تريد إضافتها، وسأقوم بتزويدك بالكود الكامل والتعديلات اللازمة مع الحفاظ على كل شيء.
